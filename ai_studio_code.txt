import { useState, useRef } from 'react';
import { QRCodeSVG } from 'qrcode.react';

export default function App() {
  const [destinationUrl, setDestinationUrl] = useState('');
  const [qrId, setQrId] = useState('');
  const [updateId, setUpdateId] = useState('');
  const [updateUrl, setUpdateUrl] = useState('');
  const [updateStatus, setUpdateStatus] = useState<'idle' | 'success' | 'error'>('idle');
  const [updateMessage, setUpdateMessage] = useState('');
  const qrRef = useRef<HTMLDivElement>(null);

  const createQr = async () => {
    const response = await fetch('/api/qr/create', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ destination_url: destinationUrl }),
    });
    const data = await response.json();
    setQrId(data.id);
  };

  const updateQr = async () => {
    setUpdateStatus('idle');
    setUpdateMessage('');
    const response = await fetch('/api/qr/update', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ id: updateId, destination_url: updateUrl }),
    });
    
    if (response.ok) {
      setUpdateStatus('success');
      setUpdateMessage('QR code updated successfully!');
    } else {
      setUpdateStatus('error');
      setUpdateMessage('Failed to update QR code. Please check the QR ID.');
    }
  };

  const downloadQr = () => {
    const svg = qrRef.current?.querySelector('svg');
    if (!svg) return;
    const svgData = new XMLSerializer().serializeToString(svg);
    const blob = new Blob([svgData], { type: "image/svg+xml;charset=utf-8" });
    const url = URL.createObjectURL(blob);
    const link = document.createElement("a");
    link.href = url;
    link.download = `qr-code-${qrId}.svg`;
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
  };

  return (
    <div className="p-8 max-w-2xl mx-auto">
      <h1 className="text-3xl font-bold mb-8">Dynamic QR Code Generator</h1>
      
      <div className="mb-8 p-4 border rounded-xl">
        <h2 className="text-xl font-semibold mb-4">Create New QR Code</h2>
        <input
          type="text"
          value={destinationUrl}
          onChange={(e) => setDestinationUrl(e.target.value)}
          placeholder="Enter destination URL"
          className="w-full p-2 border rounded mb-2"
        />
        {destinationUrl && (
          <div className="mb-4 inline-block transition-transform duration-300 hover:scale-105 hover:shadow-lg p-2 rounded-xl">
            <p className="text-sm text-gray-500 mb-1">Live Preview:</p>
            <QRCodeSVG value={destinationUrl} size={128} />
          </div>
        )}
        <button onClick={createQr} className="bg-blue-500 text-white p-2 rounded block">Create</button>
        {qrId && (
          <div className="mt-4" ref={qrRef}>
            <div className="flex items-center gap-2">
              <p className="font-mono bg-gray-100 p-1 rounded">QR ID: {qrId}</p>
              <button 
                onClick={() => navigator.clipboard.writeText(qrId)}
                className="text-xs bg-gray-200 hover:bg-gray-300 p-1 rounded"
              >
                Copy ID
              </button>
            </div>
            <div className="flex items-center gap-2 mt-2">
              <a href={`${window.location.origin}/qr/${qrId}`} target="_blank" rel="noreferrer" className="text-blue-600 underline font-mono text-sm break-all">
                {`${window.location.origin}/qr/${qrId}`}
              </a>
              <button 
                onClick={() => navigator.clipboard.writeText(`${window.location.origin}/qr/${qrId}`)}
                className="text-xs bg-gray-200 hover:bg-gray-300 p-1 rounded"
              >
                Copy Link
              </button>
            </div>
            <div className="inline-block transition-transform duration-300 hover:scale-105 hover:shadow-lg p-2 rounded-xl mt-2">
              <QRCodeSVG value={`${window.location.origin}/qr/${qrId}`} />
            </div>
            <button onClick={downloadQr} className="mt-2 bg-gray-500 text-white p-2 rounded block">Download SVG</button>
          </div>
        )}
      </div>

      <div className="p-4 border rounded-xl">
        <h2 className="text-xl font-semibold mb-4">Update Existing QR Code</h2>
        <input
          type="text"
          value={updateId}
          onChange={(e) => setUpdateId(e.target.value)}
          placeholder="Enter QR ID"
          className="w-full p-2 border rounded mb-2"
        />
        <input
          type="text"
          value={updateUrl}
          onChange={(e) => setUpdateUrl(e.target.value)}
          placeholder="Enter new destination URL"
          className="w-full p-2 border rounded mb-2"
        />
        <button onClick={updateQr} className="bg-green-500 text-white p-2 rounded">Update</button>
        {updateMessage && (
          <p className={`mt-2 ${updateStatus === 'success' ? 'text-green-600' : 'text-red-600'}`}>
            {updateMessage}
          </p>
        )}
      </div>
    </div>
  );
}
